## ARQUIVO 1 — Reescrever `src/middleware/rls.ts`

```typescript
/**
 * rls.ts — Row Level Security no Express
 * Garante que cada usuário só vê dados do próprio condomínio.
 * Fonte de verdade: req.user.condominio_id (vem do JWT assinado no login)
 */
import { Request, Response, NextFunction } from "express";

// ── Verifica que o condo acessado é o do usuário ──────────────────────────────
export function requireSameCondominio(req: Request, res: Response, next: NextFunction) {
  const user = req.user;
  if (!user) return res.status(401).json({ error: "Não autenticado" });
  if (user.role === "master") return next(); // master vê tudo

  // Extrai condominio_id de QUALQUER fonte da request
  const condoRequisitado =
    req.params.condoId      ||    // /api/gestor/condo/:condoId/...
    req.params.condominio_id ||   // /api/modulos/:condominio_id/...
    req.body?.condominio_id  ||   // body JSON
    req.query?.condominio_id as string; // query string ?condominio_id=X

  // Se veio explicitamente um condo diferente do do usuário → bloqueia
  if (condoRequisitado && condoRequisitado !== user.condominio_id) {
    return res.status(403).json({
      error: "Acesso negado: este condomínio não pertence ao seu perfil",
    });
  }
  next();
}

// ── Factory: verifica que a unidade acessada é a do morador ───────────────────
// Uso: r.get("/rota", requireSameUnidade("unidade_id"), handler)
export function requireSameUnidade(campoQuery: string = "unidade_id") {
  return (req: Request, res: Response, next: NextFunction) => {
    const user = req.user;
    if (!user) return res.status(401).json({ error: "Não autenticado" });
    if (["master", "gestor", "sindico"].includes(user.role)) return next();

    const unidadeRequisitada =
      req.params[campoQuery]   ||
      req.body?.[campoQuery]    ||
      req.query?.[campoQuery] as string;

    if (unidadeRequisitada && unidadeRequisitada !== user.unidade_id) {
      return res.status(403).json({
        error: "Acesso negado: esta unidade não pertence ao seu perfil",
      });
    }
    next();
  };
}

// ── Verifica que o usuário está ativo ─────────────────────────────────────────
// Usado em rotas críticas para bloquear usuários desativados pelo master
export function requireAtivo(req: Request, res: Response, next: NextFunction) {
  const user = req.user;
  if (!user) return res.status(401).json({ error: "Não autenticado" });
  // O campo ativo é injetado no JWT durante o login
  if ((user as any).ativo === false) {
    return res.status(403).json({ error: "Usuário inativo. Contate o administrador." });
  }
  next();
}

// ── Injeta condominio_id do JWT em todas as queries (helper) ──────────────────
// Uso: const condoId = getCondoId(req) — nunca aceitar condoId do cliente
export function getCondoId(req: Request): string {
  if (!req.user) throw new Error("Usuário não autenticado");
  return req.user.condominio_id;
}
```

r.patch("/usuarios/:id/status", async (req, res) => {
  const { ativo } = req.body;
  if (typeof ativo !== "boolean") return res.status(400).json({ error: "ativo deve ser boolean" });
  try {
    await getDB().from("usuarios")
      .update({ ativo, updated_at: new Date().toISOString() })
      .eq("id", req.params.id);

    log.info(`Master ${acao} usuário ${req.params.id}`);
