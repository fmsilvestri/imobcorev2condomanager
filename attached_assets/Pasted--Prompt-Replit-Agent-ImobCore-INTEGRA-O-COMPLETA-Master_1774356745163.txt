## PASSO 2 — Criar `src/di-engine/context.ts`

Este arquivo é a FONTE ÚNICA de verdade para o system prompt da Di.
Todos os endpoints (gestor, síndico, morador, módulos) chamam esta função.

```typescript
/**
 * di-engine/context.ts
 * Carrega o system prompt completo da Di para qualquer perfil.
 * Ordem: di_configuracoes (operacional) + di_system_prompt (blocos) + di_regras_condo
 * Master edita → banco → esta função lê → Claude recebe
 */
import { getDB } from "../db.js";
import type { Snapshot } from "../types.js";

export type Perfil = "master"|"gestor"|"sindico"|"morador"|"zelador";

export interface DiCtx {
  systemPrompt:     string;
  limiteFinanceiro: number;
  nomeDi:           string;
  tomComunicacao:   string;
  modulosAtivos:    string[];
}

// Restrições que a Di aplica por perfil (injetadas no system prompt)
const RESTRICOES_PERFIL: Record<string, string> = {
  master:  "Acesso total. Pode discutir e exibir qualquer dado.",
  gestor:  "Fala com o GESTOR. Pode discutir financeiro, OSs, aprovações Di, dados completos do condo.",
  sindico: "Fala com o SÍNDICO. NUNCA revele: saldo bancário exato, lista de inadimplentes por unidade, dados pessoais de moradores.",
  morador: "Fala com o MORADOR da unidade {{unidade_id}}. Responda APENAS sobre dados desta unidade. NUNCA mencione outros moradores, saldos do condo, detalhes de segurança.",
  zelador: "Fala com o ZELADOR. Foque em OSs operacionais, portaria e tarefas físicas. Sem dados financeiros.",
};

export async function carregarContextoDi(
  condoId:     string,
  snapshot:    Snapshot,
  perfil:      Perfil = "gestor",
  nomeUsuario: string = "Usuário",
  unidadeId?:  string
): Promise<DiCtx> {
  const db = getDB();

  // ── 1. Configurações operacionais do condo ──────────────────────────────────
  let limiteFinanceiro = 500;
  let nomeDi           = "Di";
  let tomComunicacao   = "direto_empatico";
  let modulosAtivos    = ["os","financeiro","iot","misp","encomendas","portaria",
                          "reservas","comunicados","votacoes"];
  try {
    const { data: cfg } = await db
      .from("di_configuracoes")
      .select("limite_financeiro, nome_di, tom_comunicacao, modulos_ativos")
      .eq("condominio_id", condoId)
      .single();
    if (cfg) {
      limiteFinanceiro = cfg.limite_financeiro ?? 500;
      nomeDi           = cfg.nome_di           ?? "Di";
      tomComunicacao   = cfg.tom_comunicacao   ?? "direto_empatico";
      modulosAtivos    = Array.isArray(cfg.modulos_ativos)
        ? cfg.modulos_ativos : modulosAtivos;
    }
  } catch {}

  // ── 2. Blocos do system prompt: global + custom mesclados ───────────────────
  let blocos: Array<{bloco:string; conteudo:string}> = [];
  try {
    const [g, c] = await Promise.all([
      db.from("di_system_prompt").select("bloco,conteudo")
        .is("condominio_id", null).order("bloco"),
      db.from("di_system_prompt").select("bloco,conteudo")
        .eq("condominio_id", condoId).order("bloco"),
    ]);
    const gMap = new Map((g.data??[]).map((b:any) => [b.bloco, b.conteudo]));
    const cMap = new Map((c.data??[]).map((b:any) => [b.bloco, b.conteudo]));
    // custom sobrescreve global bloco a bloco
    blocos = [...gMap.keys()].map(k => ({
      bloco: k, conteudo: cMap.get(k) ?? gMap.get(k) ?? "",
    }));
    for (const [k, v] of cMap.entries())
      if (!gMap.has(k)) blocos.push({ bloco: k, conteudo: v });
    blocos.sort((a, b) => a.bloco.localeCompare(b.bloco));
  } catch {}

  // ── 3. Regras personalizadas do condo ───────────────────────────────────────
  let regrasExtra = "";
  try {
    const { data: rg } = await db.from("di_regras_condo")
      .select("texto").eq("condominio_id", condoId)
      .eq("ativo", true).order("ordem");
    if ((rg ?? []).length > 0) {
      regrasExtra = "\nREGRAS ADICIONAIS DESTE CONDOMÍNIO:\n" +
        (rg ?? []).map((r:any, i:number) => `${i+1}. ${r.texto}`).join("\n");
    }
  } catch {}

  // ── 4. Variáveis de template ─────────────────────────────────────────────────
  const condo = snapshot.condominio as Record<string,unknown> ?? {};
  const fin   = snapshot.financeiro?.[0];
  const rv    = snapshot.reservatorios ?? [];
  const osU   = (snapshot.os_abertas ?? []).filter(o => o.prioridade === "urgente").length;

  const vars: Record<string, string> = {
    "{{nome_condominio}}":   String(condo.nome ?? "Condomínio"),
    "{{nome_gestor}}":       perfil === "gestor" ? nomeUsuario : "Gestor",
    "{{nome_sindico}}":      perfil === "sindico" ? nomeUsuario : "Síndico",
    "{{nome_usuario}}":      nomeUsuario,
    "{{unidade_id}}":        unidadeId ?? "N/D",
    "{{total_unidades}}":    String(condo.total_unidades ?? 0),
    "{{total_moradores}}":   String(condo.total_moradores ?? 0),
    "{{timestamp}}":         new Date().toLocaleString("pt-BR"),
    "{{os_abertas}}":        String((snapshot.os_abertas ?? []).length),
    "{{os_urgentes}}":       String(osU),
    "{{saldo}}":             fin ? Number(fin.saldo_atual).toLocaleString("pt-BR") : "N/D",
    "{{inadimplencia_pct}}": fin ? String(fin.inadimplencia_percentual) : "N/D",
    "{{score_geral}}":       String((snapshot.scores ?? [])[0]?.score ?? "N/D"),
    "{{misp_ativos}}":       String((snapshot.alertas_misp ?? []).length),
    "{{limite_financeiro}}": String(limiteFinanceiro),
    "{{limite}}":            String(limiteFinanceiro),
    "{{tom_comunicacao}}":   tomComunicacao.replace(/_/g, " "),
    "{{reservatorios}}":     rv.map(r => `${r.nome}:${r.ultimo_nivel_pct?.toFixed(0)}%`)
                               .join(", ") || "N/D",
    "{{aprovacoes}}":        String((snapshot.acoes_pendentes ?? []).length),
  };

  const sub = (txt: string): string => {
    let r = txt;
    for (const [k, v] of Object.entries(vars)) r = r.replaceAll(k, v);
    return r;
  };

  // ── 5. Montar system prompt ──────────────────────────────────────────────────
  let systemPrompt: string;

  if (blocos.length > 0) {
    // Para chat: exclui bloco 6_formato (que é JSON — chat usa linguagem natural)
    const blocosChat = blocos.filter(b => b.bloco !== "6_formato");
    systemPrompt = blocosChat
      .map(b => sub(b.conteudo))
      .filter(Boolean)
      .join("\n\n---\n\n");
    if (regrasExtra) systemPrompt += "\n\n" + regrasExtra;
  } else {
    // Fallback se banco não tem blocos
    systemPrompt = `Você é ${nomeDi}, Síndica Virtual do condomínio ${vars["{{nome_condominio}}"]}.
Tom: ${vars["{{tom_comunicacao}}"]}.
Limite financeiro autônomo: R$ ${limiteFinanceiro}.
Nunca aja acima deste limite sem aprovação do gestor.`;
  }

  // ── 6. Estado em tempo real + restrição por perfil ──────────────────────────
  const restricao = sub(
    RESTRICOES_PERFIL[perfil] ?? RESTRICOES_PERFIL.morador
  );

  systemPrompt += `\n\n---\n\nESTADO ATUAL (${vars["{{timestamp}}"]}):
- Condomínio: ${vars["{{nome_condominio}}"]} | ${vars["{{total_unidades}}"]} unidades
- OSs: ${vars["{{os_abertas}}"]} abertas (${vars["{{os_urgentes}}"]} urgentes)
- Financeiro: Saldo R$${vars["{{saldo}}"]} | Inadimp ${vars["{{inadimplencia_pct}}"]}%
- Reservatórios: ${vars["{{reservatorios}}"]}
- Score: ${vars["{{score_geral}}"]}/100 | MISP: ${vars["{{misp_ativos}}"]} | Aprovações pendentes: ${vars["{{aprovacoes}}"]}
- Módulos ativos: ${modulosAtivos.join(", ")}

PERFIL ATUAL: ${restricao}`;

  return {
    systemPrompt,
    limiteFinanceiro,
    nomeDi,
    tomComunicacao,
    modulosAtivos,
  };
}
```

/**
 * routes/modulos.ts — /api/modulos/*
 * Rota unificada para todos os perfis.
 * Permissões e comportamento Da Di definidos pelo Master via di_configuracoes + di_system_prompt.
 */
import { Router, Request, Response } from "express";
import { requireAuth }           from "../middleware/auth.js";
import { requireSameCondominio } from "../middleware/rls.js";
import { getDB }                 from "../db.js";
import { gerarSnapshot }         from "../services/snapshot.js";
import { carregarContextoDi }    from "../di-engine/context.js";
import { MODULOS, getModulosVisíveis, getCampos } from "../di-engine/modulos.js";
import Anthropic from "@anthropic-ai/sdk";
import { CONFIG } from "../config.js";
import type { Perfil } from "../di-engine/modulos.js";

const r = Router();
r.use(requireAuth(["master","gestor","sindico","morador","zelador"]));
r.use(requireSameCondominio);

let _cl: Anthropic | null = null;
const cl = () => _cl ??= new Anthropic({ apiKey: CONFIG.anthropic.apiKey });

// ── Listar módulos visíveis para o perfil ─────────────────────────────────────
r.get("/", async (req: Request, res: Response) => {
  const condoId = req.user!.condominio_id;
  const perfil  = (req.user!.role ?? "morador") as Perfil;
  try {
    const { data: cfg } = await getDB()
      .from("di_configuracoes").select("modulos_ativos")
      .eq("condominio_id", condoId).single();
    const ids: string[] = cfg?.modulos_ativos ?? [
      "os","financeiro","iot","encomendas","portaria","reservas","comunicados","votacoes",
    ];
    const mods = getModulosVisíveis(ids, perfil).map(m => ({
      id: m.id, nome: m.nome, icone: m.icone,
      pode_editar:  m.editar.includes(perfil),
      pode_aprovar: (m.aprovar ?? []).includes(perfil),
    }));
    res.json({ modulos: mods, perfil, total: mods.length });
  } catch { res.json({ modulos: [], perfil, total: 0 }); }
});

// ── Dados do módulo filtrados por perfil ──────────────────────────────────────
r.get("/:id/dados", async (req: Request, res: Response) => {
  const { id }   = req.params;
  const condoId  = req.user!.condominio_id;
  const perfil   = (req.user!.role ?? "morador") as Perfil;
  const unidade  = req.user!.unidade_id;
  const mod      = MODULOS[id];
  if (!mod) return res.status(404).json({ error: "Módulo não encontrado" });
  if (!mod.ver.includes(perfil)) return res.status(403).json({ error: "Sem permissão" });
  const campos = getCampos(mod, perfil);
  if (!campos) return res.json({ dados: [], modulo: id, perfil });
  try {
    const db = getDB();
    let q = db.from(mod.tabela).select(campos).eq("condominio_id", condoId);
    // Isolamento morador
    if (perfil === "morador" && unidade) {
      if (id === "encomendas") q = q.eq("destinatario_unidade", unidade);
      if (id === "portaria")   q = q.eq("destino_unidade", unidade);
      if (id === "reservas")   q = q.eq("unidade_id", unidade);
    }
    if (mod.filtros?.status) q = q.in("status", mod.filtros.status);
    if (id === "iot")      q = q.eq("ativo", true);
    if (id === "portaria") q = q.gte("created_at", new Date(Date.now() - 86400000).toISOString());
    if (id === "reservas") q = q.gte("data", new Date().toISOString().split("T")[0]);
    const { data } = await q.order("created_at", { ascending: false }).limit(mod.limite);
    res.json({ dados: data ?? [], modulo: id, perfil, total: (data ?? []).length });
  } catch (e) { res.json({ dados: [], modulo: id, perfil, error: (e as Error).message }); }
});

// ── Di analisa módulo ─────────────────────────────────────────────────────────
r.post("/:id/di-analise", async (req: Request, res: Response) => {
  const { id }   = req.params;
  const condoId  = req.user!.condominio_id;
  const perfil   = (req.user!.role ?? "morador") as Perfil;
  const nome     = req.user!.nome ?? perfil;
  const unidade  = req.user!.unidade_id;
  const mod      = MODULOS[id];
  if (!mod) return res.status(404).json({ error: "Módulo não encontrado" });
  if (!mod.ver.includes(perfil)) return res.status(403).json({ error: "Sem permissão" });
  try {
    // Buscar dados filtrados
    const campos = getCampos(mod, perfil);
    let dados: Record<string, unknown>[] = [];
    if (campos) {
      const db = getDB();
      let q = db.from(mod.tabela).select(campos).eq("condominio_id", condoId);
      if (perfil === "morador" && unidade) {
        if (id === "encomendas") q = q.eq("destinatario_unidade", unidade);
        if (id === "portaria")   q = q.eq("destino_unidade", unidade);
        if (id === "reservas")   q = q.eq("unidade_id", unidade);
      }
      if (mod.filtros?.status) q = q.in("status", mod.filtros.status);
      if (id === "iot") q = q.eq("ativo", true);
      if (id === "portaria") q = q.gte("created_at", new Date(Date.now()-86400000).toISOString());
      const { data } = await q.order("created_at", { ascending: false }).limit(mod.limite);
      dados = (data ?? []) as Record<string, unknown>[];
    }

    // System prompt personalizado do Master
    const snapshot = await gerarSnapshot(condoId);
    const ctx      = await carregarContextoDi(condoId, snapshot, perfil, nome, unidade);

    // Injeta instrução específica do módulo + restrições por perfil
    const pp = mod.tomPorPerfil[perfil];
    const sysMod = `${ctx.systemPrompt}

---

MÓDULO ATIVO: ${mod.nome} (${mod.icone})
${mod.promptDi}
${pp ? `
CONTEXTO PARA ${perfil.toUpperCase()}: ${pp.contexto}
TOM: ${pp.tom}
RESTRIÇÕES ABSOLUTAS PARA ESTE PERFIL: ${pp.restricoes}
` : ""}

Responda em linguagem natural (não JSON neste endpoint).
Formato: 3 parágrafos. 1. Status. 2. Atenção. 3. Recomendação.`;

    const msg = await cl().messages.create({
      model: CONFIG.anthropic.model, max_tokens: 500,
      system: sysMod,
      messages: [{
        role: "user",
        content: `Dados de ${mod.nome}:\n${JSON.stringify(dados, null, 2).slice(0, 3000)}\n\nAnalise para ${perfil} ${nome}.`,
      }],
    });

    const analise = (msg.content.find(b => b.type === "text") as Anthropic.TextBlock)?.text ?? "";
    res.json({
      modulo: id, nome: mod.nome, icone: mod.icone,
      perfil, analise_di: analise, dados,
      tokens: msg.usage.input_tokens + msg.usage.output_tokens,
      timestamp: new Date().toISOString(),
    });
  } catch {
    res.json({
      modulo: id, perfil,
      analise_di: `Di processando ${mod.nome}.`,
      dados: [], offline: true,
    });
  }
});

// ── Chat Di contextualizado por módulo ────────────────────────────────────────
r.post("/:id/di-chat", async (req: Request, res: Response) => {
  const { id } = req.params;
  const { mensagem, historico = [] } = req.body ?? {};
  if (!mensagem) return res.status(400).json({ error: "mensagem obrigatória" });
  const condoId = req.user!.condominio_id;
  const perfil  = (req.user!.role ?? "morador") as Perfil;
  const nome    = req.user!.nome ?? perfil;
  const mod     = MODULOS[id];
  if (!mod) return res.status(404).json({ error: "Módulo não encontrado" });
  try {
    const snapshot = await gerarSnapshot(condoId);
    const ctx      = await carregarContextoDi(condoId, snapshot, perfil, nome, req.user!.unidade_id);
    const pp       = mod.tomPorPerfil[perfil];
    const sys      = `${ctx.systemPrompt}\n\n---\n\nMÓDULO: ${mod.nome}\n${mod.promptDi}${pp ? `\nTOM: ${pp.tom}\nRESTRIÇÕES: ${pp.restricoes}` : ""}`;
    const messages: Anthropic.MessageParam[] = [
      ...historico.slice(-6).map((m: { role: string; content: string }) => ({
        role: m.role as "user"|"assistant", content: m.content,
      })),
      { role: "user" as const, content: mensagem },
    ];
    const msg = await cl().messages.create({
      model: CONFIG.anthropic.model, max_tokens: 400, system: sys, messages,
    });
    const resposta = (msg.content.find(b => b.type === "text") as Anthropic.TextBlock)?.text ?? "";
    res.json({ resposta, perfil, modulo: id });
  } catch {
    res.json({ resposta: "Estou aqui para ajudar.", offline: true });
  }
});

export default r;
