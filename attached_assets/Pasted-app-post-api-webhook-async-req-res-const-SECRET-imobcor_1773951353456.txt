app.post('/api/webhook', async (req, res) => {
  const SECRET = "imobcore-webhook-secret";

  if (req.headers['x-webhook-secret'] !== SECRET) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  const { device_id, distancia_cm, nivel_percent, volume_litros, bateria, raw } = req.body;

  if (!device_id) {
    return res.status(400).json({ error: 'device_id ausente' });
  }

  try {
    // 1. buscar reservoir pelo iot_sensor_id
    const { data: reservoir, error: rErr } = await supabase
      .from('reservoirs')
      .select('id, name, capacity_liters')
      .eq('iot_sensor_id', device_id)
      .single();

    if (rErr || !reservoir) {
      console.error('Reservoir não encontrado:', device_id, rErr);
      return res.status(404).json({ error: 'Reservoir not found', device_id });
    }

    // 2. calcular volume se não vier preenchido
    let volFinal = volume_litros;
    let nivelFinal = nivel_percent;

    if (!volFinal && distancia_cm && reservoir.capacity_liters) {
      // ESP32 envia distância do sensor até a água — inverter para nível
      // distancia_cm já vem dividido por 100 no Worker
      const alturaAgua = 200 - distancia_cm; // 200cm = height_cm padrão
      const pct = Math.max(0, Math.min(alturaAgua / 200, 1));
      volFinal = Math.round(pct * reservoir.capacity_liters);
      nivelFinal = Math.round(pct * 100);
    }

    // 3. INSERT em sensor_readings (colunas confirmadas)
    const { error: insErr } = await supabase
      .from('sensor_readings')
      .insert({
        condominium_id: reservoir.condominium_id ?? req.body.condominium_id,
        reservoir_id:   reservoir.id,
        device_id:      device_id,
        nivel:          nivelFinal,
        distancia:      distancia_cm,
        volume:         volFinal,
        raw_payload:    raw ?? req.body,
        created_at:     new Date().toISOString()
      });

    if (insErr) {
      console.error('Erro INSERT sensor_readings:', insErr);
      return res.status(500).json({ error: insErr.message });
    }

    // 4. atualizar iot_last_reading e iot_last_sync no reservoir
    await supabase
      .from('reservoirs')
      .update({
        iot_last_reading: nivelFinal,
        iot_last_sync:    new Date().toISOString(),
        iot_status:       'online'
      })
      .eq('id', reservoir.id);

    // 5. broadcast SSE
    broadcastSSE({
      type:         'water_reading',
      reservoir_id: reservoir.id,
      device_id,
      nivel:        nivelFinal,
      volume:       volFinal,
      timestamp:    new Date().toISOString()
    });

    console.log('✅ Leitura salva:', reservoir.name, nivelFinal + '%', volFinal + 'L');
    return res.json({ ok: true, reservoir_id: reservoir.id, nivel: nivelFinal, volume: volFinal });

  } catch (err) {
    console.error('Webhook erro geral:', err);
    return res.status(500).json({ error: err.message });
  }
});
