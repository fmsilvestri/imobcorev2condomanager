app.post("/api/di", async (req, res) => {

  const dados = await carregarDados(); // supabase + iot

  const resumo = gerarResumo(dados);

  const cards = gerarCards(dados);

  res.json({
    fala: resumo,
    cards
  });
});

function gerarResumo(d){

  let texto = "Olá 👋 estou monitorando o condomínio.\n";

  if(d.agua.nivel < 25){
    texto += "O nível da água está crítico. ";
  }

  if(d.financeiro.inadimplencia > 10){
    texto += "A inadimplência está acima do ideal. ";
  }

  if(d.manutencao.urgentes > 0){
    texto += "Há manutenções urgentes pendentes. ";
  }

  return texto;
}

function gerarCards(d){

  return [
    {
      titulo:"💧 Água",
      valor: d.agua.nivel + "%",
      cor: d.agua.nivel < 25 ? "red":"green"
    },
    {
      titulo:"💰 Financeiro",
      valor: d.financeiro.inadimplencia + "%",
      cor: d.financeiro.inadimplencia > 10 ? "yellow":"green"
    },
    {
      titulo:"🛠️ Manutenção",
      valor: d.manutencao.abertos,
      cor: d.manutencao.urgentes > 0 ? "red":"green"
    }
  ];
}

function falar(texto, setFalando){

  const u = new SpeechSynthesisUtterance(texto);
  u.lang = "pt-BR";

  u.onstart = () => setFalando(true);
  u.onend = () => setFalando(false);

  speechSynthesis.speak(u);
}
📊 5. CARDS INTELIGENTES COLORIDOS
function Card({titulo, valor, cor}){

  const cores = {
    green:"bg-green-500",
    yellow:"bg-yellow-500",
    red:"bg-red-500"
  };

  return (
    <div className={`p-4 rounded-xl text-white ${cores[cor]}`}>
      <h3>{titulo}</h3>
      <p className="text-xl font-bold">{valor}</p>
    </div>
  );
}
🔁 6. DASHBOARD COMPLETO COM DI
export default function Dashboard(){

  const [cards,setCards] = useState([]);
  const [fala,setFala] = useState("");
  const [falando,setFalando] = useState(false);

  async function atualizar(){

    const r = await fetch("/api/di", { method:"POST" });
    const data = await r.json();

    setCards(data.cards);
    setFala(data.fala);

    falar(data.fala, setFalando);
  }

  useEffect(()=>{
    atualizar();
    setInterval(atualizar, 10000);
  },[]);

  return (
    <div className="bg-black text-white min-h-screen p-4">

      <DiAvatar falando={falando} />

      <div className="mt-4 bg-gray-800 p-3 rounded">
        {fala}
      </div>

      <div className="grid grid-cols-2 gap-3 mt-6">
        {cards.map((c,i)=><Card key={i} {...c}/>)}
      </div>

    </div>
  );
}
