```sql
CREATE OR REPLACE FUNCTION fn_gerar_token_totem()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO di_configuracoes (condominio_id, concierge_token)
  VALUES (NEW.id, encode(gen_random_bytes(24),'hex'))
  ON CONFLICT (condominio_id) DO UPDATE
    SET concierge_token = COALESCE(
      di_configuracoes.concierge_token,
      encode(gen_random_bytes(24),'hex')
    );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

  AFTER INSERT ON condominios
  FOR EACH ROW EXECUTE FUNCTION fn_gerar_token_totem();
```

```typescript
/**
 * services/totem.ts
 * Isolamento total: token → condoId → todas queries filtradas.
 * Nenhuma função retorna dados sem filtrar por condoId.

import { getDB }         from '../db.js';
import { gerarSnapshot } from './snapshot.js';

export interface TotemCtx {
  condoId:     string;
  nomeCondo:   string;
  nomeDi:      string;
  corTema:     string;
  ttsProv:     string;
  idleSeg:     number;
  saudacao:    string;
  horarios:    Record<string,string>;
  contatos:    Record<string,string>;
  regras:      string;
  avatarUrl:   string;
  wifiRede:    string;
  wifiSenha:   string;
  checkinH:    string;
  checkoutH:   string;
  cidade:      string;
  faq:         Array<{pergunta:string;resposta:string;categoria:string}>;
  midia:       Array<{id:string;titulo:string;descricao:string;categoria:string;
                       tipo:string;url:string;slideshow:boolean;carousel:boolean;ordem:number}>;
}

/** Valida token e retorna condoId. Lança erro se inválido ou desativado. */
export async function resolveToken(token: string): Promise<string> {
  if (!token?.trim()) throw new Error('Token ausente');
  const { data, error } = await getDB()
    .from('di_configuracoes')
    .select('condominio_id,concierge_ativo')
    .eq('concierge_token', token.trim())
    .single();
  if (error || !data)        throw new Error('Token inválido');
  if (!data.concierge_ativo) throw new Error('Totem desativado para este condomínio');
  return data.condominio_id as string;
}

/** Carrega contexto COMPLETO do condo em paralelo. Nunca mistura condos. */
export async function carregarCtx(condoId: string): Promise<TotemCtx> {
  const [cfgR, condoR, faqR, midiaR] = await Promise.allSettled([
    getDB().from('di_configuracoes')
      .select('nome_di,concierge_saudacao,concierge_cor_tema,concierge_tts_provider,'+
              'concierge_idle_seg,concierge_horarios,concierge_contatos,concierge_regras,'+
              'concierge_avatar_url,concierge_wifi_rede,concierge_wifi_senha,'+
              'concierge_checkin_h,concierge_checkout_h,concierge_cidade')
      .eq('condominio_id', condoId).single(),
    getDB().from('condominios').select('nome').eq('id', condoId).single(),
    getDB().from('concierge_faq').select('pergunta,resposta,categoria')
      .eq('condominio_id', condoId).eq('ativo', true).order('ordem').limit(25),
    getDB().from('totem_midia')
      .select('id,titulo,descricao,categoria,tipo,url,slideshow,carousel,ordem')
      .eq('condominio_id', condoId).eq('ativo', true).order('ordem'),
  ]);
  const cfg   = cfgR.status   === 'fulfilled' ? cfgR.value.data   : null;
  const condo = condoR.status === 'fulfilled' ? condoR.value.data : null;
  const faq   = faqR.status   === 'fulfilled' ? faqR.value.data ?? [] : [];
  const midia = midiaR.status === 'fulfilled' ? midiaR.value.data ?? [] : [];
  const nomeDi    = cfg?.nome_di    ?? 'Di';
  const nomeCondo = condo?.nome     ?? 'Condomínio';
  return {
    condoId, nomeCondo, nomeDi,
    corTema:    cfg?.concierge_cor_tema     ?? '#7c3aed',
    ttsProv:    cfg?.concierge_tts_provider ?? 'web_speech',
    idleSeg:    cfg?.concierge_idle_seg     ?? 90,
    saudacao:   (cfg?.concierge_saudacao ?? `Olá! Sou ${nomeDi}, síndica de ${nomeCondo}.`)
                  .replace('{nome_di}', nomeDi).replace('{nome_condo}', nomeCondo),
    horarios:   (cfg?.concierge_horarios  ?? {}) as Record<string,string>,
    contatos:   (cfg?.concierge_contatos  ?? {portaria:'',emergencia:'192'}) as Record<string,string>,
    regras:     cfg?.concierge_regras     ?? '',
    avatarUrl:  cfg?.concierge_avatar_url ?? '',
    wifiRede:   cfg?.concierge_wifi_rede  ?? '',
    wifiSenha:  cfg?.concierge_wifi_senha ?? '',
    checkinH:   cfg?.concierge_checkin_h  ?? '14h',
    checkoutH:  cfg?.concierge_checkout_h ?? '12h',
    cidade:     cfg?.concierge_cidade     ?? 'Rio de Janeiro',
    faq:   faq   as TotemCtx['faq'],
    midia: midia as TotemCtx['midia'],
  };
}

/** Guia agrupado por categoria — filtrado por condoId. */
export async function carregarGuia(condoId: string): Promise<Record<string,any[]>> {
  const { data } = await getDB().from('guia_hospede')
    .select('id,categoria,titulo,descricao,icone,conteudo,horarios,regras,dicas,'+
            'tags,telefones,localizacao,pontos,comercios,desconto,tipo_parceria,status,destaque')
    .eq('condominio_id', condoId).eq('ativo', true).order('ordem');
  return (data ?? []).reduce((acc: Record<string,any[]>, it) => {
    const c = it.categoria ?? 'info';
    if (!acc[c]) acc[c] = [];
    acc[c].push(it);
    return acc;
  }, {});
}

/** System prompt rico com snapshot REAL daquele condo + config Master. */
export async function montarSystemPrompt(ctx: TotemCtx, tipo: string): Promise<string> {
  let snap: any = {};
  try { snap = await gerarSnapshot(ctx.condoId); } catch {}
  const os     = snap.os_abertas    ?? [];
  const misp   = snap.alertas_misp  ?? [];
  const encom  = snap.encomendas    ?? [];
  const scores = snap.scores        ?? [];
  const reserv = snap.reservatorios ?? [];
  const cInfo  = (snap.condominio   ?? {}) as Record<string,any>;
  const osUrg     = os.filter((o:any)  => o.prioridade === 'urgente').length;
  const mispUrg   = misp.filter((m:any) => m.severidade === 'urgente').length;
  const scoreGeral= scores.length ? Math.round(scores.reduce((s:number,p:any)=>s+(p.score??0),0)/scores.length) : null;
  const nivelAgua = reserv.length ? Math.round(reserv.reduce((s:number,r:any)=>s+(r.ultimo_nivel_pct??0),0)/reserv.length) : null;

  const CTX: Record<string,string> = {
    hospede:    'HÓSPEDE buscando informações. Responda sobre áreas, serviços, horários, '+
                'passeios em '+ctx.cidade+', parceiros com descontos. Nunca revele dados de moradores.',
    visitante:  'VISITANTE na portaria. Colete nome completo e apartamento destino. '+
                'Ao obter ambos, confirme notificação ao morador.',
    entrega:    'ENTREGADOR. Colete empresa/remetente e apartamento. Confirme o registro.',
    morador:    'MORADOR. Auxilie com reservas, comunicados e serviços.',
    emergencia: 'EMERGÊNCIA. Forneça contatos imediatamente. Tom calmo e direto.',
    info:       'INFORMAÇÕES GERAIS. Use horários e FAQ para orientar.',
  };

  const horStr = Object.keys(ctx.horarios).length
    ? '\nHORÁRIOS:\n'+Object.entries(ctx.horarios).map(([k,v])=>`• ${k}: ${v}`).join('\n') : '';
  const wifiStr = ctx.wifiRede
    ? `\nWI-FI: Rede "${ctx.wifiRede}" | Senha: ${ctx.wifiSenha||'consultar portaria'}` : '';

  return `Você é ${ctx.nomeDi}, síndica virtual e concierge do ${ctx.nomeCondo} em ${ctx.cidade}.
Está num TOTEM na entrada. Responda em PT-BR, cordial, máx 3 frases.

${CTX[tipo] ?? CTX.info}

━━ ${ctx.nomeCondo.toUpperCase()} — ESTADO REAL AGORA ━━
Unidades: ${cInfo.total_unidades??'—'} | Moradores: ${cInfo.total_moradores??'—'}
OSs abertas: ${os.length} (${osUrg} urgentes)
Alertas MISP: ${misp.length} (${mispUrg} urgentes)
Encomendas aguardando: ${encom.length}
${scoreGeral!==null?`Score geral: ${scoreGeral}/100`:''}
${nivelAgua!==null?`Nível reservatórios: ${nivelAgua}%`:''}

━━ CONFIG DESTE CONDOMÍNIO ━━
${horStr}${wifiStr}${checkStr}
${ctx.regras?'\nREGRAS:\n'+ctx.regras:''}
${faqStr}

━━ REGRAS ABSOLUTAS ━━
• NUNCA revele CPF, e-mail ou dados pessoais de moradores
• NUNCA informe saldo, inadimplência ou finanças internas
• Emergência: portaria ${ctx.contatos.portaria||'ver portaria'} | ${ctx.contatos.emergencia||'192'}
• Se não souber: oriente a falar com o porteiro`;
}
```

```typescript
// ── Middleware auth por token ──────────────────────────────────────────────────
async function auth(req: Request, res: Response, next: Function) {
  const token = (req.headers['x-concierge-token'] as string)
             || (req.query.token as string)
             || (CONFIG.nodeEnv === 'development' ? process.env.CONCIERGE_TOKEN : undefined);
  if (!token) return res.status(401).json({
    error: 'Token obrigatório. Configure o tablet com a URL do totem do seu condomínio.'
  });
  try {
    (req as any).condoId = await resolveToken(token);
    (req as any).token   = token;
    next();
  } catch (e) { res.status(403).json({ error: (e as Error).message }); }
}

  const { mensagem, historico = [], tipo = 'info' } = req.body ?? {};
  if (!mensagem?.trim()) return res.status(400).json({ error: 'mensagem obrigatória' });
  const condoId = (req as any).condoId;
  try {
    const ctx = await carregarCtx(condoId);
    const systemPrompt = await montarSystemPrompt(ctx, tipo);
    const messages: Anthropic.MessageParam[] = [
      ...historico.slice(-8).map((m: any) => ({ role: m.role as 'user'|'assistant', content: m.content })),
      { role: 'user' as const, content: mensagem },
    ];
    const msg = await cl().messages.create({
      model: CONFIG.anthropic.model, max_tokens: 300, system: systemPrompt, messages,
    });
    const resposta = (msg.content.find(b => b.type === 'text') as Anthropic.TextBlock)?.text ?? '';
    // Ações automáticas
    let acao = false;
    if (tipo === 'visitante') acao = await _regVisitante(mensagem, historico, condoId);
    if (tipo === 'entrega')   acao = await _regEntrega(mensagem, historico, condoId);
    if (tipo === 'emergencia') _regEmergencia(condoId).catch(() => {});
    getDB().from('concierge_sessoes').insert({ condominio_id: condoId, tipo }).catch(() => {});
    res.json({ resposta, acao_executada: acao });
  } catch { res.json({ resposta: 'Não foi possível conectar. Fale com a portaria.', offline: true }); }
});

async function _regVisitante(msg:string, hist:any[], condoId:string): Promise<boolean> {
  const conv = [...hist.map((m:any)=>m.content), msg].join(' ');
  const apto = conv.match(/\b(?:ap(?:to)?\.?\s*|apartamento\s*)(\d{2,4}[a-z]?)/i)?.[1];
  const nome = conv.match(/(?:me chamo|meu nome [eé]|sou o?a?\s+)([A-ZÁÀÂÃÉÊÍÓÔÕÚÇ][a-záàâãéêíóôõúç]+(?:\s+[A-Z][a-z]+)*)/i)?.[1];
  if (!apto || !nome) return false;
  await getDB().from('visitantes_portaria')
    .insert({ condominio_id:condoId, nome, destino_unidade:apto, tipo:'visita', status:'aguardando_autorizacao' })
    .catch(()=>{});
  return true;
}
async function _regEntrega(msg:string, hist:any[], condoId:string): Promise<boolean> {
  const conv = [...hist.map((m:any)=>m.content), msg].join(' ').toLowerCase();
  const apto = conv.match(/\b(?:ap(?:to)?\.?\s*|apartamento\s*)(\d{2,4}[a-z]?)/i)?.[1];
  if (!apto) return false;
  const rem = ['amazon','correios','ifood','rappi','shein','shopee'].find(x=>conv.includes(x)) ?? 'Entregador';
  const perecivel = ['ifood','rappi','uber'].some(x=>conv.includes(x));
  await getDB().from('encomendas').insert({
    condominio_id:condoId, remetente:rem, tipo:perecivel?'perecivel':'normal',
    destinatario_unidade:apto, sla_horas:perecivel?4:168,
    status:'aguardando_retirada', lembretes_enviados:0,
  }).catch(()=>{});
  return true;
}
async function _regEmergencia(condoId:string): Promise<void> {
  await getDB().from('alertas_misp').insert({
    condominio_id:condoId, tipo:'EMERGENCIA_TOTEM',
    descricao:'Emergência acionada no Concierge Virtual — entrada',
    local:'Totem entrada', severidade:'urgente', status:'ativo',
  });
}
```

```typescript
/** routes/tts.ts — ElevenLabs Jenifer com cache em memória */
import { Router } from 'express';
const r = Router();
const cache = new Map<string, Buffer>();

r.post('/', async (req, res) => {
  const { text, idioma = 'pt_BR' } = req.body ?? {};
  if (!text) return res.status(400).json({ error: 'text obrigatório' });
  const txt     = String(text).slice(0, 500);
  const apiKey  = process.env.ELEVENLABS_API_KEY;
  const voiceId = process.env.ELEVENLABS_VOICE_ID;
  if (!apiKey || !voiceId)
    return res.status(503).json({ error: 'ElevenLabs não configurado — adicione ELEVENLABS_API_KEY e ELEVENLABS_VOICE_ID nos Secrets' });
  const key = `${idioma}:${txt}`;
  if (cache.has(key)) { res.setHeader('Content-Type','audio/mpeg'); return res.end(cache.get(key)); }
  try {
    const resp = await fetch(`https://api.elevenlabs.io/v1/text-to-speech/${voiceId}/stream`,{
      method:'POST',
      headers:{'xi-api-key':apiKey,'Content-Type':'application/json','Accept':'audio/mpeg'},
      body:JSON.stringify({text:txt,model_id:'eleven_multilingual_v2',
        voice_settings:{stability:0.45,similarity_boost:0.82,style:0.35,use_speaker_boost:true}}),
    });
    if (!resp.ok) return res.status(502).json({ error: 'ElevenLabs ' + resp.status });
    res.setHeader('Content-Type','audio/mpeg');
    const chunks: Buffer[] = [];
    const reader = resp.body!.getReader();
    const pump = async (): Promise<void> => {
      const { done, value } = await reader.read();
      if (done) { if (cache.size < 60) cache.set(key, Buffer.concat(chunks)); res.end(); return; }
      const c = Buffer.from(value); chunks.push(c); res.write(c); return pump();
    };
    await pump();
  } catch (e) { res.status(502).json({ error: (e as Error).message }); }
});

export default r;
```

```javascript
async function salvarTotem() {
  const condoId = S.selectedCondoId;
  await api(`/master/di/settings/${condoId}`, { method:'PUT', body:JSON.stringify({
    concierge_ativo:        document.getElementById('tog-totem').classList.contains('on'),
    concierge_saudacao:     document.getElementById('totem-saud').value,
    concierge_cidade:       document.getElementById('totem-cidade').value,

      emergencia: document.getElementById('totem-emg').value,

async function totemRegenToken() {
  if (!confirm('Gerar novo token?\n\nO link atual deixará de funcionar — atualize o tablet.')) return;
  const newToken = Array.from(crypto.getRandomValues(new Uint8Array(24)))
    .map(b=>b.toString(16).padStart(2,'0')).join('');
  await api(`/master/di/settings/${S.selectedCondoId}`,
    { method:'PUT', body:JSON.stringify({ concierge_token: newToken }) });
  showToast('Novo token gerado! Atualize o tablet com a nova URL.');
  loadTotemTab(S.selectedCondoId);
}

  const url = document.getElementById('totem-url').textContent.trim();
  if (url && url !== '—') window.open(url, '_blank');
}

}
```

| O que            | Isolado por         | Query exemplo                                         |
|------------------|---------------------|-------------------------------------------------------|
| Config Di        | condominio_id       | `.eq('condominio_id', condoId)`                       |
| Guia/Áreas       | condominio_id       | `.eq('condominio_id', condoId).eq('ativo', true)`     |
| FAQ              | condominio_id       | `.eq('condominio_id', condoId).eq('ativo', true)`     |
| Galeria mídia    | condominio_id       | `.eq('condominio_id', condoId).eq('ativo', true)`     |
| Snapshot Di      | condominioId        | `gerarSnapshot(condoId)` → todas queries com `.eq()`  |
| Visitantes       | condominio_id       | `INSERT { condominio_id: condoId, ... }`               |
| Entregas         | condominio_id       | `INSERT { condominio_id: condoId, ... }`               |
| Emergências      | condominio_id       | `INSERT { condominio_id: condoId, ... }`               |
| Métricas sessões | condominio_id       | `.eq('condominio_id', condoId).gte('iniciada_em',...)`|
| Token            | UNIQUE no Supabase  | Cada token resolve exatamente 1 condoId               |
