## VISÃO GERAL DA NOVA ARQUITETURA

```
Master Panel
  └─ Condo A → Tab "🖥 Totem" → token: a3f9c1... → URL: /totem/a3f9c1...
  └─ Condo B → Tab "🖥 Totem" → token: b7d2e4... → URL: /totem/b7d2e4...
  └─ Condo C → Tab "🖥 Totem" → token: c5f8a2... → URL: /totem/c5f8a2...

Tablet portaria Condo A abre: https://app.repl.co/totem/a3f9c1...
  └─ Backend: resolveToken("a3f9c1...") → condoId_A
  └─ HTML injetado com window.TOTEM_TOKEN e window.TOTEM_CONDO_ID
  └─ Todas as chamadas API usam o token → dados EXCLUSIVOS do Condo A

Di no Condo A responde com:
  • nome e saudação personalizados do Condo A
  • snapshot real: OSs, alertas, scores, reservatórios do Condo A
  • FAQ, horários, regras configurados pelo Master para o Condo A
  • NUNCA acessa dados do Condo B ou C

ISOLAMENTO GARANTIDO:
  • Token → condoId → todas queries filtram por .eq("condominio_id", condoId)
  • Token inválido = 403 imediato
  • Token desativado = 403 imediato
  • Dados de condos diferentes NUNCA se misturam
```

-- 2. Trigger: garante que todo condo novo receba token automaticamente
-- ═══════════════════════════════════════════════════════════════════
CREATE OR REPLACE FUNCTION fn_gerar_token_totem()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO di_configuracoes (condominio_id, concierge_token)
  VALUES (NEW.id, encode(gen_random_bytes(24), 'hex'))
  ON CONFLICT (condominio_id) DO UPDATE
    SET concierge_token = COALESCE(
      EXCLUDED.concierge_token,
      encode(gen_random_bytes(24), 'hex')
    );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_token_totem ON condominios;
CREATE TRIGGER trg_token_totem
  AFTER INSERT ON condominios
  FOR EACH ROW EXECUTE FUNCTION fn_gerar_token_totem();

/**
 * services/totem.ts
 * Serviço central de isolamento do Totem.
 * Toda operação começa com resolveToken() → condoId.
 * Nenhuma query retorna dados sem filtrar por condoId.
 */
import { getDB }         from '../db.js';
import { gerarSnapshot } from './snapshot.js';

// ─── Tipos ───────────────────────────────────────────────────────────────────
export interface TotemCtx {
  condoId:     string;
  nomeCondo:   string;
  nomeDi:      string;
  corTema:     string;
  ttsProv:     string;
  idleSeg:     number;
  saudacao:    string;
  horarios:    Record<string,string>;
  contatos:    Record<string,string>;
  regras:      string;
  avatarUrl:   string;
  faq:         Array<{pergunta:string;resposta:string;categoria:string}>;
  midia:       Array<{id:string;titulo:string;descricao:string;categoria:string;
                       tipo:string;url:string;slideshow:boolean;carousel:boolean;ordem:number}>;
}

// ─── resolveToken ─────────────────────────────────────────────────────────────
// Valida o token e retorna o condoId. Lança erro em qualquer falha.
export async function resolveToken(token: string): Promise<string> {
  if (!token?.trim()) throw new Error('Token ausente');

  const { data, error } = await getDB()
    .from('di_configuracoes')
    .select('condominio_id, concierge_ativo')
    .eq('concierge_token', token.trim())
    .single();

  if (error || !data)        throw new Error('Token inválido');
  if (!data.concierge_ativo) throw new Error('Totem desativado para este condomínio');

  return data.condominio_id as string;
}

// ─── carregarCtx ─────────────────────────────────────────────────────────────
// Carrega TODO o contexto personalizado daquele condo em paralelo.
export async function carregarCtx(condoId: string): Promise<TotemCtx> {
  const [cfgR, condoR, faqR, midiaR] = await Promise.allSettled([
    getDB().from('di_configuracoes')
      .select('nome_di,concierge_saudacao,concierge_cor_tema,concierge_tts_provider,'+
              'concierge_idle_seg,concierge_horarios,concierge_contatos,'+
              'concierge_regras,concierge_avatar_url')
      .eq('condominio_id', condoId).single(),

    getDB().from('condominios')
      .select('nome')
      .eq('id', condoId).single(),

    getDB().from('concierge_faq')
      .select('pergunta,resposta,categoria')
      .eq('condominio_id', condoId).eq('ativo', true)
      .order('ordem').limit(25),

    getDB().from('totem_midia')
      .select('id,titulo,descricao,categoria,tipo,url,slideshow,carousel,ordem')
      .eq('condominio_id', condoId).eq('ativo', true)
      .order('ordem'),
  ]);

  const cfg   = cfgR.status   === 'fulfilled' ? cfgR.value.data   : null;
  const condo = condoR.status === 'fulfilled' ? condoR.value.data : null;
  const faq   = faqR.status   === 'fulfilled' ? faqR.value.data ?? [] : [];
  const midia = midiaR.status === 'fulfilled' ? midiaR.value.data ?? [] : [];

  const nomeDi    = cfg?.nome_di    ?? 'Di';
  const nomeCondo = condo?.nome     ?? 'Condomínio';

  const saudacaoBase = cfg?.concierge_saudacao
    ?? `Olá! Sou ${nomeDi}, síndica virtual de ${nomeCondo}.`;

  return {
    condoId,
    nomeCondo,
    nomeDi,
    corTema:   cfg?.concierge_cor_tema      ?? '#7c3aed',
    ttsProv:   cfg?.concierge_tts_provider  ?? 'web_speech',
    idleSeg:   cfg?.concierge_idle_seg      ?? 90,
    saudacao:  saudacaoBase
               .replace('{nome_di}',   nomeDi)
               .replace('{nome_condo}', nomeCondo),
    horarios:  (cfg?.concierge_horarios ?? {}) as Record<string,string>,
    contatos:  (cfg?.concierge_contatos ?? {portaria:'',emergencia:'192'}) as Record<string,string>,
    regras:    cfg?.concierge_regras   ?? '',
    avatarUrl: cfg?.concierge_avatar_url ?? '',
    faq:   faq   as TotemCtx['faq'],
    midia: midia as TotemCtx['midia'],
  };
}

// ─── carregarGuia ─────────────────────────────────────────────────────────────
export async function carregarGuia(condoId: string): Promise<Record<string,any[]>> {
  const { data } = await getDB().from('guia_hospede')
    .select('id,categoria,titulo,descricao,icone,conteudo,horarios,regras,dicas,'+
            'telefones,localizacao,pontos,comercios,desconto,tipo_parceria,destaque')
    .eq('condominio_id', condoId).eq('ativo', true).order('ordem');

  return (data ?? []).reduce((acc: Record<string,any[]>, it) => {
    const c = it.categoria ?? 'info';
    if (!acc[c]) acc[c] = [];
    acc[c].push(it);
    return acc;
  }, {});
}

// Constrói o system prompt RICO com dados reais do snapshot + config do condo.
export async function montarSystemPrompt(ctx: TotemCtx, tipo: string): Promise<string> {
  // Snapshot real — filtrado pelo condoId do contexto
  let snap: any = {};
  try { snap = await gerarSnapshot(ctx.condoId); } catch { /* usa {} */ }

  const os     = snap.os_abertas    ?? [];
  const misp   = snap.alertas_misp  ?? [];
  const encom  = snap.encomendas    ?? [];

  const mispUrg    = misp.filter((m:any) => m.severidade === 'urgente').length;

/**
 * routes/concierge.ts
 * API do Totem — auth por token, isolamento total por condoId.
 * Nenhum endpoint vaza dados entre condomínios.
 */
import { Router, Request, Response } from 'express';
import { getDB }          from '../db.js';
import { requireMaster }  from '../middleware/auth.js';
import {
  resolveToken, carregarCtx, carregarGuia,
  montarSystemPrompt, seedGuiaDemo,
} from '../services/totem.js';
import Anthropic from '@anthropic-ai/sdk';
import { CONFIG } from '../config.js';

const r = Router();
let _cl: Anthropic | null = null;
const cl = () => _cl ??= new Anthropic({ apiKey: CONFIG.anthropic.apiKey });

// ── auth middleware ────────────────────────────────────────────────────────────
async function auth(req: Request, res: Response, next: Function) {
  const token = (req.headers['x-concierge-token'] as string)
             || (req.query.token as string)
             || (CONFIG.nodeEnv === 'development' ? process.env.CONCIERGE_TOKEN : undefined);

  if (!token) return res.status(401).json({
    error: 'Token obrigatório. Configure o tablet com a URL do totem do seu condomínio.',
  });

  try {
    (req as any).condoId = await resolveToken(token);
    (req as any).token   = token;
    next();
  } catch (e) {
    res.status(403).json({ error: (e as Error).message });
  }
}
r.use(auth);

// ═══════════════════════════════════════════════════════════════════
// GET /api/concierge/config  — inicialização do totem
// ═══════════════════════════════════════════════════════════════════
r.get('/config', async (req: Request, res: Response) => {
  try {
    const ctx = await carregarCtx((req as any).condoId);
    res.json({
      condominio_id: ctx.condoId,
      nome_condo:    ctx.nomeCondo,
      nome_di:       ctx.nomeDi,
      cor_tema:      ctx.corTema,
      tts_provider:  ctx.ttsProv,
      idle_seg:      ctx.idleSeg,
      saudacao:      ctx.saudacao,
      horarios:      ctx.horarios,
      contatos:      ctx.contatos,
      avatar_url:    ctx.avatarUrl,
      faq:           ctx.faq,
      midia:         ctx.midia,
    });
  } catch (e) { res.status(500).json({ error: (e as Error).message }); }
});

// ═══════════════════════════════════════════════════════════════════
// POST /api/concierge/chat  — Di com Claude + snapshot real
// ═══════════════════════════════════════════════════════════════════
r.post('/chat', async (req: Request, res: Response) => {
  const { mensagem, historico = [], tipo = 'info' } = req.body ?? {};
  if (!mensagem?.trim()) return res.status(400).json({ error: 'mensagem obrigatória' });

  const condoId = (req as any).condoId;

  try {
    const ctx          = await carregarCtx(condoId);
    const systemPrompt = await montarSystemPrompt(ctx, tipo);

    const messages: Anthropic.MessageParam[] = [
      ...historico.slice(-8).map((m: any) => ({
        role: m.role as 'user'|'assistant', content: m.content,
      })),
      { role: 'user' as const, content: mensagem },
    ];

    const msg = await cl().messages.create({
      model: CONFIG.anthropic.model, max_tokens: 300,
      system: systemPrompt, messages,
    });

    const resposta = (msg.content.find(b => b.type === 'text') as Anthropic.TextBlock)?.text ?? '';

    // Ações automáticas — sempre isoladas por condoId
    let acao = false;
    if (tipo === 'visitante') acao = await _regVisitante(mensagem, historico, condoId);
    if (tipo === 'entrega')   acao = await _regEntrega(mensagem, historico, condoId);
    if (tipo === 'emergencia') _regEmergencia(condoId).catch(() => {});

    // Log (não bloqueia resposta)
    getDB().from('concierge_sessoes')
      .insert({ condominio_id: condoId, tipo })
      .then(() => {}).catch(() => {});

    res.json({ resposta, acao_executada: acao });

  } catch {
    res.json({ resposta: 'Não foi possível conectar. Fale com a portaria.', offline: true });
  }
});

async function _regVisitante(msg:string, hist:any[], condoId:string): Promise<boolean> {
  const conv = [...hist.map((m:any) => m.content), msg].join(' ');
  const apto = conv.match(/\b(?:ap(?:to)?\.?\s*|apartamento\s*)(\d{2,4}[a-z]?)/i)?.[1];
  const nome = conv.match(/(?:me chamo|meu nome [eé]|sou o?a?\s+)([A-ZÁÀÂÃÉÊÍÓÔÕÚÇ][a-záàâãéêíóôõúç]+(?:\s+[A-Z][a-z]+)*)/i)?.[1];
  if (!apto || !nome) return false;
  await getDB().from('visitantes_portaria')
    .insert({condominio_id:condoId, nome, destino_unidade:apto,
             tipo:'visita', status:'aguardando_autorizacao'}).catch(()=>{});
  await getDB().from('notificacoes')
    .insert({condominio_id:condoId, titulo:`Visitante: ${nome}`,
             corpo:`${nome} chegou para o Apto ${apto}.`,
             canal:'push', destino:apto, lida:false}).catch(()=>{});
  return true;
}
async function _regEntrega(msg:string, hist:any[], condoId:string): Promise<boolean> {
  const conv = [...hist.map((m:any) => m.content), msg].join(' ').toLowerCase();
  const apto = conv.match(/\b(?:ap(?:to)?\.?\s*|apartamento\s*)(\d{2,4}[a-z]?)/i)?.[1];
  if (!apto) return false;
  const rem = ['amazon','correios','ifood','rappi','shein','shopee','mercado livre']
    .find(x => conv.includes(x)) ?? 'Entregador';
  const perecivel = ['ifood','rappi','uber'].some(x => conv.includes(x));
  await getDB().from('encomendas').insert({
    condominio_id:condoId, remetente:rem,
    tipo:perecivel?'perecivel':'normal', destinatario_unidade:apto,
    sla_horas:perecivel?4:168, status:'aguardando_retirada', lembretes_enviados:0,
  }).catch(()=>{});
  return true;
}

  });

```typescript
/**
 * routes/tts.ts — ElevenLabs Jenifer com cache em memória
 */
import { Router } from 'express';
const r = Router();
const cache = new Map<string, Buffer>();

r.post('/', async (req, res) => {
  const { text, idioma = 'pt_BR' } = req.body ?? {};
  if (!text) return res.status(400).json({ error: 'text obrigatório' });

  const txt     = String(text).slice(0, 500);
  const apiKey  = process.env.ELEVENLABS_API_KEY;
  const voiceId = process.env.ELEVENLABS_VOICE_ID;

  if (!apiKey || !voiceId)
    return res.status(503).json({ error: 'ElevenLabs não configurado. Adicione ELEVENLABS_API_KEY e ELEVENLABS_VOICE_ID nos Secrets do Replit.' });

  const key = `${idioma}:${txt}`;
  if (cache.has(key)) {
    res.setHeader('Content-Type', 'audio/mpeg');
    return res.end(cache.get(key));
  }

  try {
    const resp = await fetch(
      `https://api.elevenlabs.io/v1/text-to-speech/${voiceId}/stream`,
      {
        method:  'POST',
        headers: {
          'xi-api-key':   apiKey,
          'Content-Type': 'application/json',
          'Accept':       'audio/mpeg',
        },
        body: JSON.stringify({
          text: txt,
          model_id: 'eleven_multilingual_v2',
          voice_settings: { stability:0.45, similarity_boost:0.82, style:0.35, use_speaker_boost:true },
        }),
      }
    );
    if (!resp.ok) return res.status(502).json({ error: 'ElevenLabs ' + resp.status });

    res.setHeader('Content-Type', 'audio/mpeg');
    const chunks: Buffer[] = [];
    const reader = resp.body!.getReader();
    const pump = async (): Promise<void> => {
      const { done, value } = await reader.read();
      if (done) {
        if (cache.size < 60) cache.set(key, Buffer.concat(chunks));
        res.end(); return;
      }
      const c = Buffer.from(value);
      chunks.push(c); res.write(c);
      return pump();
    };
    await pump();
  } catch (e) { res.status(502).json({ error: (e as Error).message }); }
});

export default r;
```

async function loadTotemTab(condoId) {

function _renderHors() {
  document.getElementById('totem-hor-list').innerHTML =

async function salvarTotem() {
  const condoId = S.selectedCondoId;
  await api(`/master/di/settings/${condoId}`, {
    method: 'PUT',
    body: JSON.stringify({
      concierge_ativo:        document.getElementById('tog-totem').classList.contains('on'),
      concierge_saudacao:     document.getElementById('totem-saud').value,
      concierge_cor_tema:     document.getElementById('totem-cor').value,
      concierge_tts_provider: document.getElementById('totem-tts').value,
      concierge_idle_seg:     Number(document.getElementById('totem-idle').value)||90,
      concierge_regras:       document.getElementById('totem-regras').value,
      concierge_horarios:     _totemHors,
      concierge_contatos: {
        portaria:   document.getElementById('totem-port').value,
        emergencia: document.getElementById('totem-emg').value,
      },
    }),
  });
  await api(`/api/concierge/master/faq/${condoId}`, {
    method: 'PUT',
    body: JSON.stringify({ faq: _totemFaq }),
  });
  showToast(`✓ Totem de ${S.selectedCondoNm} salvo!`);
}

async function totemRegenToken() {
  if (!confirm(
    'Gerar novo token?\n\n' +
    'A URL atual deixará de funcionar e o tablet precisará ser reconfigurado.'
  )) return;
  const newToken = Array.from(crypto.getRandomValues(new Uint8Array(24)))
    .map(b => b.toString(16).padStart(2,'0')).join('');
  await api(`/master/di/settings/${S.selectedCondoId}`, {
    method: 'PUT',
    body: JSON.stringify({ concierge_token: newToken }),
  });
  showToast('Novo token gerado! Atualize o tablet com a nova URL.');
  loadTotemTab(S.selectedCondoId);
}

function totemCopyUrl() {
  const url = document.getElementById('totem-url').textContent.trim();
  navigator.clipboard.writeText(url).then(() =>
    showToast('✓ URL copiada! Cole no navegador do tablet.')
  );
}

function totemAbrir() {
  const url = document.getElementById('totem-url').textContent.trim();
  if (url && url !== '—') window.open(url, '_blank');
}
