```typescript
// Calcular próxima execução:
function calcProxima(plano: PlanoManutencao): Date {
  const base = plano.ultima_execucao ? new Date(plano.ultima_execucao) : new Date(plano.created_at)
  const dias = plano.frequencia_tipo === 'mensal' ? 30 * plano.frequencia_valor
             : plano.frequencia_tipo === 'semanal' ? 7 * plano.frequencia_valor
             : plano.frequencia_tipo === 'semestral' ? 180
             : plano.frequencia_tipo === 'anual' ? 365 : 30
  return new Date(base.getTime() + dias * 86400000)
}

// Status de urgência da próxima execução:
function urgencia(proxima: Date): 'vencido' | 'urgente' | 'ok' {
  const diff = (proxima.getTime() - Date.now()) / 86400000
  if(diff < 0) return 'vencido'
  if(diff <= 7) return 'urgente'
  return 'ok'
}
```

### `src/components/Planos/CalendarioAnual.tsx` — Calendário visual

```typescript
// Timeline horizontal de 12 meses com eventos de manutenção
// Calcular automaticamente todas as datas de execução de todos os planos ativos
// Código de cores: verde (concluído), âmbar (vencendo em 7d), azul (agendado), cinza (anual)
// Grid de custo mensal estimado (12 cards com R$ de cada mês)
// Botão exportar PDF do calendário → chamar generateCalendarioPDF()

function gerarEventosCalendario(planos: PlanoManutencao[]): CalendarioEvent[] {
  const eventos: CalendarioEvent[] = []
  const hoje = new Date()
  
  planos.forEach(plano => {
    const diasFreq = plano.frequencia_tipo === 'mensal' ? 30 * plano.frequencia_valor
                   : plano.frequencia_tipo === 'semanal' ? 7 * plano.frequencia_valor
                   : plano.frequencia_tipo === 'semestral' ? 180
                   : plano.frequencia_tipo === 'anual' ? 365 : 30
    
    let data = new Date(hoje.getFullYear(), 0, 1) // 01/jan do ano atual
    while(data.getFullYear() === hoje.getFullYear()) {
      eventos.push({
        data: new Date(data),
        planoId: plano.id,
        planoNome: plano.nome,
        setor: plano.setor,
        custo: plano.custo_estimado
      })
      data = new Date(data.getTime() + diasFreq * 86400000)
    }
  })
  
  return eventos.sort((a,b) => a.data.getTime() - b.data.getTime())
}
```

```typescript
async function createFromPlano(plano: PlanoManutencao): Promise<OrdemServico> {
  const proxima = calcProxima(plano)
  const numero = await generateNumero(plano.condominio_id)
  
  const os: Partial<OrdemServico> = {
    condominio_id: plano.condominio_id,
    numero,
    titulo: plano.nome,
    categoria: plano.setor,
    status: 'aberta',
    prioridade: 'media',
    responsavel: plano.prestador_nome,
    prestador_nome: plano.prestador_nome,
    custo_estimado: plano.custo_estimado,
    data_prevista: proxima.toISOString().split('T')[0],
    sla_horas: 48,
    checklist: (plano.template_checklist as any[] || []).map(c => 
      typeof c === 'string' ? {item: c, done: false} : c
    ),
    aprovacao_necessaria: (plano.custo_estimado || 0) > 500
  }
  
  const {data, error} = await supabase.from('ordens_servico').insert([os]).select().single()
  if(error) throw error
  
  // Atualizar ultima_execucao no plano
  await supabase.from('planos_manutencao')
    .update({ultima_execucao: new Date().toISOString(), execucoes_realizadas: (plano.execucoes_realizadas || 0) + 1})
    .eq('id', plano.id)
  
  return data
}
```

## 4. CRON JOB — Supabase Edge Function (geração automática)

Criar `supabase/functions/gerar-os-automaticas/index.ts`:

```typescript
// Executar diariamente via cron (ou Cloudflare Workers)
// Lógica:
// 1. Buscar todos os planos ativos com gerar_os_automatica = true
// 2. Para cada plano, calcular próxima execução
// 3. Se próxima execução <= hoje + dias_antecedencia → criar OS automaticamente
// 4. Evitar duplicatas verificando se já existe OS aberta/em andamento para este plano
// 5. Notificar síndico via realtime channel

Deno.serve(async () => {
  const supabase = createClient(Deno.env.get('SUPABASE_URL')!, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!)
  
  const {data: planos} = await supabase
    .from('planos_manutencao')
    .select('*')
    .eq('ativo', true)
    .eq('gerar_os_automatica', true)
  
  for(const plano of planos || []) {
    const proxima = calcProxima(plano)
    const diasRestantes = (proxima.getTime() - Date.now()) / 86400000
    
    if(diasRestantes <= plano.dias_antecedencia) {
      // Verificar se já existe OS aberta para este plano
      const {data: existing} = await supabase
        .from('ordens_servico')
        .select('id')
        .eq('condominio_id', plano.condominio_id)
        .eq('titulo', plano.nome)
        .in('status', ['aberta', 'andamento'])
        .single()
      
      if(!existing) {
        await createFromPlano(plano, supabase)
      }
    }
  }
  
  return new Response('OK')
})
```
