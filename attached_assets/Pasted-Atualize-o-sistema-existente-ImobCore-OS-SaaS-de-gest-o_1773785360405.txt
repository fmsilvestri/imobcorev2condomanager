export function calcularScore(respostas) {

let total = 0
let max = 0

const categorias = {
financeiro: 0,
manutencao: 0,
operacao: 0,
iot: 0
}

respostas.forEach(r => {
max += r.peso

```
if (r.resposta === true) {
  total += r.peso
  categorias[r.categoria] += r.peso
}
```

})

const scoreTotal = Math.round((total / max) * 100)

return {
scoreTotal,
categorias
}

}

export function gerarInsights(score) {

const insights = []

if (score.financeiro < 60) {
insights.push({
tipo: "financeiro",
mensagem: "Inadimplência elevada detectada",
prioridade: "alta"
})
}

if (score.manutencao < 60) {
insights.push({
tipo: "manutencao",
mensagem: "Falta de manutenção preventiva",
prioridade: "alta"
})
}

if (score.operacao < 60) {
insights.push({
tipo: "operacao",
mensagem: "Existem ordens de serviço atrasadas",
prioridade: "media"
})
}

return insights

}
