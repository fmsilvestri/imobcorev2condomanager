📲 3. TELEGRAM (RÁPIDO E GRATUITO)
Criar bot no Telegram:

👉 fale com Telegram → @BotFather

Código:
const TELEGRAM_TOKEN = "SEU_TOKEN";
const CHAT_ID = "ID_DO_GRUPO";

export async function enviarTelegram(msg){

  await fetch(`https://api.telegram.org/bot${TELEGRAM_TOKEN}/sendMessage`, {
    method:"POST",
    headers:{ "Content-Type":"application/json" },
    body: JSON.stringify({
      chat_id: CHAT_ID,
      text: msg
    })
  });

}

    method:"POST",
    headers:{
      "Authorization":"Bearer SEU_TOKEN",
      "Content-Type":"application/json"
    },

      to: telefone,

  });

}
📱 5. PUSH NOTIFICATION (APP)

Use:

👉 Expo Notifications

Backend:
export async function enviarPush(token, msg){

  await fetch("https://exp.host/--/api/v2/push/send",{
    method:"POST",
    headers:{ "Content-Type":"application/json" },
    body: JSON.stringify({
      to: token,
      sound:"default",
      body: msg
    })
  });

}
