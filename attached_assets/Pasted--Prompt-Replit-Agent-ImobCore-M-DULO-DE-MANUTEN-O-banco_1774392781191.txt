```sql
-- ── Equipamentos e estruturas do condomínio ───────────────────────────────────
CREATE TABLE IF NOT EXISTS equipamentos (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  condominio_id     UUID REFERENCES condominios(id) ON DELETE CASCADE,
  nome              TEXT NOT NULL,
  categoria         TEXT NOT NULL, -- elevador|bomba|gerador|cftv|portao|climatizacao|piscina|estrutura|outro
  local             TEXT,          -- Bloco A, Subsolo, Cobertura...
  fabricante        TEXT,
  modelo            TEXT,
  numero_serie      TEXT,
  data_instalacao   DATE,
  vida_util_anos    INTEGER DEFAULT 10,
  status            TEXT DEFAULT 'operacional', -- operacional|atencao|critico|inativo|em_manutencao
  custo_manutencao  NUMERIC DEFAULT 0,          -- custo acumulado
  proxima_manutencao DATE,
  intervalo_manutencao_dias INTEGER DEFAULT 90,
  qr_code           TEXT UNIQUE,                -- identificador único para QR
  foto_url          TEXT,
  observacoes       TEXT,
  created_at        TIMESTAMPTZ DEFAULT NOW(),
  updated_at        TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS eq_condo     ON equipamentos(condominio_id, status);
CREATE INDEX IF NOT EXISTS eq_prox_manu ON equipamentos(proxima_manutencao, condominio_id);
CREATE INDEX IF NOT EXISTS eq_qrcode    ON equipamentos(qr_code);

-- ── Histórico de manutenções ──────────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS manutencoes (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  condominio_id   UUID REFERENCES condominios(id) ON DELETE CASCADE,
  equipamento_id  UUID REFERENCES equipamentos(id) ON DELETE CASCADE,
  tipo            TEXT NOT NULL, -- preventiva|corretiva|emergencial|inspecao
  titulo          TEXT NOT NULL,
  descricao       TEXT,
  status          TEXT DEFAULT 'agendada', -- agendada|em_andamento|concluida|cancelada
  prioridade      TEXT DEFAULT 'normal',   -- urgente|alta|normal|baixa
  tecnico_nome    TEXT,
  tecnico_contato TEXT,
  custo_estimado  NUMERIC,
  custo_real      NUMERIC,
  os_vinculada_id UUID REFERENCES ordens_servico(id),  -- link com OS existente
  agendada_para   TIMESTAMPTZ,
  concluida_em    TIMESTAMPTZ,
  proxima_em      TIMESTAMPTZ,             -- Di calcula automaticamente
  criada_por_di   BOOLEAN DEFAULT false,   -- true = Di criou automaticamente
  notas_di        TEXT,                    -- observações da Di sobre esta manutenção
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS man_equip  ON manutencoes(equipamento_id, status);
CREATE INDEX IF NOT EXISTS man_condo  ON manutencoes(condominio_id, status, agendada_para);
CREATE INDEX IF NOT EXISTS man_agenda ON manutencoes(agendada_para) WHERE status='agendada';

-- ── Plano de manutenção (recorrências configuradas) ───────────────────────────
CREATE TABLE IF NOT EXISTS plano_manutencao (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  condominio_id     UUID REFERENCES condominios(id) ON DELETE CASCADE,
  equipamento_id    UUID REFERENCES equipamentos(id) ON DELETE CASCADE,
  tipo_servico      TEXT NOT NULL,   -- Troca de óleo, Limpeza filtros...
  intervalo_dias    INTEGER NOT NULL, -- 30, 90, 180, 365
  ultimo_realizado  DATE,
  proximo_previsto  DATE,
  responsavel       TEXT,
  custo_estimado    NUMERIC,
  ativo             BOOLEAN DEFAULT true,
  created_at        TIMESTAMPTZ DEFAULT NOW()
);

-- ── Alertas de manutenção gerados pela Di ────────────────────────────────────
CREATE TABLE IF NOT EXISTS alertas_manutencao (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  condominio_id   UUID REFERENCES condominios(id) ON DELETE CASCADE,
  equipamento_id  UUID REFERENCES equipamentos(id),
  manutencao_id   UUID REFERENCES manutencoes(id),
  tipo            TEXT NOT NULL, -- vencimento|status_critico|custo_alto|preventiva_atraso
  mensagem        TEXT NOT NULL,
  severidade      TEXT DEFAULT 'normal', -- urgente|alta|normal
  resolvido       BOOLEAN DEFAULT false,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- ── RLS ───────────────────────────────────────────────────────────────────────
ALTER TABLE equipamentos       ENABLE ROW LEVEL SECURITY;
ALTER TABLE manutencoes        ENABLE ROW LEVEL SECURITY;
ALTER TABLE plano_manutencao   ENABLE ROW LEVEL SECURITY;
ALTER TABLE alertas_manutencao ENABLE ROW LEVEL SECURITY;

-- Políticas básicas (gestor e síndico do mesmo condo)
CREATE POLICY "eq_gestor" ON equipamentos FOR ALL
  USING (condominio_id IN (
    SELECT condominio_id FROM usuarios WHERE id = auth.uid()
  ));

-- ── View: equipamentos com status calculado ───────────────────────────────────
CREATE OR REPLACE VIEW v_equipamentos_status AS
SELECT
  e.*,
  (SELECT COUNT(*) FROM manutencoes m
   WHERE m.equipamento_id = e.id AND m.status IN ('agendada','em_andamento')) AS manutencoes_ativas,
  (SELECT MAX(m.concluida_em) FROM manutencoes m
   WHERE m.equipamento_id = e.id AND m.status = 'concluida') AS ultima_manutencao,
  CASE
    WHEN e.proxima_manutencao < CURRENT_DATE THEN 'vencida'
    WHEN e.proxima_manutencao < CURRENT_DATE + 7 THEN 'proxima'
    ELSE 'ok'
  END AS status_manutencao
FROM equipamentos e;

-- ── Função: score de manutenção melhorado (substituir o atual) ────────────────
-- O scoring.ts atual usa apenas ordens_servico.
-- Com equipamentos, o score fica muito mais preciso:
-- SELECT * FROM fn_score_manutencao('condominio_uuid');
CREATE OR REPLACE FUNCTION fn_score_manutencao(p_condo UUID)
RETURNS TABLE(score INTEGER, equipamentos_criticos INTEGER, manutencoes_vencidas INTEGER,
              proximas_7d INTEGER, custo_mes NUMERIC) AS $$
BEGIN
  RETURN QUERY
  SELECT
    GREATEST(0, 100
      - (SELECT COUNT(*) FROM equipamentos WHERE condominio_id=p_condo AND status='critico')::int * 20
      - (SELECT COUNT(*) FROM equipamentos WHERE condominio_id=p_condo AND status='atencao')::int * 8
      - (SELECT COUNT(*) FROM equipamentos WHERE condominio_id=p_condo AND proxima_manutencao < CURRENT_DATE)::int * 12
      - (SELECT COUNT(*) FROM manutencoes WHERE condominio_id=p_condo AND status='agendada' AND agendada_para < NOW())::int * 5
    )::integer AS score,
    (SELECT COUNT(*) FROM equipamentos WHERE condominio_id=p_condo AND status='critico')::integer,
    (SELECT COUNT(*) FROM equipamentos WHERE condominio_id=p_condo AND proxima_manutencao < CURRENT_DATE)::integer,
    (SELECT COUNT(*) FROM equipamentos WHERE condominio_id=p_condo AND proxima_manutencao BETWEEN CURRENT_DATE AND CURRENT_DATE + 7)::integer,
    (SELECT COALESCE(SUM(custo_real),0) FROM manutencoes WHERE condominio_id=p_condo AND concluida_em >= date_trunc('month', NOW()))::numeric;
END;
$$ LANGUAGE plpgsql;
```

```typescript
/**
 * routes/manutencao.ts
 * Módulo de Manutenção com integração Da Di
 * Equipamentos, histórico, plano, QR codes, alertas automáticos
 */
import { Router, Request, Response } from "express";
import { requireAuth }           from "../middleware/auth.js";
import { requireSameCondominio } from "../middleware/rls.js";
import { getDB }                 from "../db.js";
import { gerarSnapshot }         from "../services/snapshot.js";
import { criarOS }               from "../action-engine/os.js";
import Anthropic from "@anthropic-ai/sdk";
import { CONFIG } from "../config.js";

const r = Router();
r.use(requireAuth(["gestor","sindico","zelador","master"]));
r.use(requireSameCondominio);

let _cl: Anthropic | null = null;
const cl = () => _cl ??= new Anthropic({ apiKey: CONFIG.anthropic.apiKey });

// ── EQUIPAMENTOS ─────────────────────────────────────────────────────────────

// Listar todos
r.get("/equipamentos", async (req: Request, res: Response) => {
  const condoId  = req.user!.condominio_id;
  const { status, categoria } = req.query;
  try {
    let q = getDB().from("equipamentos")
      .select("*")
      .eq("condominio_id", condoId)
      .order("proxima_manutencao", { ascending: true, nullsFirst: false });
    if (status)    q = q.eq("status", status as string);
    if (categoria) q = q.eq("categoria", categoria as string);
    const { data } = await q;
    res.json(data ?? []);
  } catch { res.json([]); }
});

// Criar equipamento (gestor/master)
r.post("/equipamentos", async (req: Request, res: Response) => {
  if (!["gestor","master"].includes(req.user!.role))
    return res.status(403).json({ error: "Apenas gestor pode criar equipamentos" });
  const condoId = req.user!.condominio_id;
  const { nome, categoria, local, fabricante, modelo, numero_serie,
          data_instalacao, vida_util_anos, intervalo_manutencao_dias,
          proxima_manutencao, custo_manutencao, observacoes } = req.body;
  if (!nome || !categoria) return res.status(400).json({ error: "nome e categoria obrigatórios" });

  // Gera QR code único
  const qr_code = `EQ-${condoId.slice(0,8).toUpperCase()}-${Date.now()}`;

  try {
    const { data, error } = await getDB().from("equipamentos").insert([{
      condominio_id: condoId, nome, categoria, local, fabricante, modelo,
      numero_serie, data_instalacao, vida_util_anos: vida_util_anos ?? 10,
      intervalo_manutencao_dias: intervalo_manutencao_dias ?? 90,
      proxima_manutencao, custo_manutencao: custo_manutencao ?? 0,
      observacoes, qr_code, status: "operacional",
      created_at: new Date().toISOString(), updated_at: new Date().toISOString(),
    }]).select().single();
    res.json({ ok: !error, data, qr_code });
  } catch (e) { res.json({ ok: false, error: (e as Error).message }); }
});

// Buscar por QR code (zelador escaneia)
r.get("/equipamentos/qr/:qrCode", async (req: Request, res: Response) => {
  const condoId = req.user!.condominio_id;
  try {
    const { data } = await getDB().from("equipamentos")
      .select("*").eq("qr_code", req.params.qrCode)
      .eq("condominio_id", condoId).single();
    if (!data) return res.status(404).json({ error: "Equipamento não encontrado" });
    // Busca histórico de manutenções
    const { data: hist } = await getDB().from("manutencoes")
      .select("*").eq("equipamento_id", data.id)
      .order("created_at", { ascending: false }).limit(5);
    res.json({ equipamento: data, historico: hist ?? [] });
  } catch { res.status(404).json({ error: "QR code inválido" }); }
});

// Atualizar status
r.patch("/equipamentos/:id/status", async (req: Request, res: Response) => {
  const { status, observacoes } = req.body;
  try {
    const { data } = await getDB().from("equipamentos")
      .update({ status, observacoes, updated_at: new Date().toISOString() })
      .eq("id", req.params.id).select().single();
    // Se status virou critico → Di cria OS automaticamente
    if (status === "critico") {
      await criarOS(req.user!.condominio_id, {
        titulo:     `Equipamento em estado crítico: ${(data as any)?.nome}`,
        prioridade: "urgente",
        area:       (data as any)?.local ?? "Geral",
        sla_horas:  4,
      });
    }
    res.json({ ok: true, data });
  } catch (e) { res.json({ ok: false, error: (e as Error).message }); }
});

// ── MANUTENÇÕES ───────────────────────────────────────────────────────────────

// Listar manutenções do condo
r.get("/historico", async (req: Request, res: Response) => {
  const condoId = req.user!.condominio_id;
  const { equipamento_id, status, tipo } = req.query;
  try {
    let q = getDB().from("manutencoes")
      .select("*, equipamentos(nome,categoria,local)")
      .eq("condominio_id", condoId)
      .order("agendada_para", { ascending: false }).limit(50);
    if (equipamento_id) q = q.eq("equipamento_id", equipamento_id as string);
    if (status)         q = q.eq("status", status as string);
    if (tipo)           q = q.eq("tipo", tipo as string);
    const { data } = await q;
    res.json(data ?? []);
  } catch { res.json([]); }
});

// Registrar manutenção
r.post("/historico", async (req: Request, res: Response) => {
  const condoId = req.user!.condominio_id;
  const { equipamento_id, tipo, titulo, descricao, status, prioridade,
          tecnico_nome, tecnico_contato, custo_estimado, agendada_para } = req.body;
  if (!equipamento_id || !titulo)
    return res.status(400).json({ error: "equipamento_id e titulo obrigatórios" });
  try {
    const { data, error } = await getDB().from("manutencoes").insert([{
      condominio_id: condoId, equipamento_id, tipo: tipo ?? "preventiva",
      titulo, descricao, status: status ?? "agendada",
      prioridade: prioridade ?? "normal",
      tecnico_nome, tecnico_contato,
      custo_estimado, agendada_para,
      created_at: new Date().toISOString(), updated_at: new Date().toISOString(),
    }]).select().single();
    res.json({ ok: !error, data });
  } catch (e) { res.json({ ok: false, error: (e as Error).message }); }
});

// Concluir manutenção
r.patch("/historico/:id/concluir", async (req: Request, res: Response) => {
  const { custo_real, notas, proxima_em } = req.body;
  try {
    const { data } = await getDB().from("manutencoes").update({
      status: "concluida",
      custo_real, notas_di: notas,
      concluida_em: new Date().toISOString(),
      proxima_em,
      updated_at: new Date().toISOString(),
    }).eq("id", req.params.id).select().single();

    // Atualizar proxima_manutencao no equipamento
    if ((data as any)?.equipamento_id && proxima_em) {
      await getDB().from("equipamentos").update({
        proxima_manutencao: proxima_em,
        updated_at: new Date().toISOString(),
      }).eq("id", (data as any).equipamento_id);
    }
    res.json({ ok: true, data });
  } catch (e) { res.json({ ok: false, error: (e as Error).message }); }
});

// ── DASHBOARD DA Di ───────────────────────────────────────────────────────────

r.get("/dashboard", async (req: Request, res: Response) => {
  const condoId = req.user!.condominio_id;
  try {

      getDB().from("equipamentos").select("id,nome,categoria,status,proxima_manutencao,custo_manutencao")

      getDB().from("plano_manutencao").select("*")
        .eq("condominio_id", condoId).eq("ativo", true)
        .lt("proximo_previsto", new Date(Date.now()+7*86400000).toISOString().split("T")[0]),
    ]);

    const eq  = equipamentos.status==="fulfilled" ? equipamentos.value.data??[] : [];
    const man = manutencoes.status==="fulfilled"  ? manutencoes.value.data??[]  : [];
    const pl  = plano.status==="fulfilled"         ? plano.value.data??[]        : [];

    const stats = {
      total_equipamentos:  eq.length,
      operacionais:        eq.filter((e:any)=>e.status==="operacional").length,
      atencao:             eq.filter((e:any)=>e.status==="atencao").length,
      criticos:            eq.filter((e:any)=>e.status==="critico").length,
      inativos:            eq.filter((e:any)=>e.status==="inativo").length,
      manutencoes_vencidas:eq.filter((e:any)=>e.proxima_manutencao && new Date(e.proxima_manutencao) < new Date()).length,
      proximas_7d:         eq.filter((e:any)=>{
        if(!e.proxima_manutencao) return false;
        const d = new Date(e.proxima_manutencao);
        return d >= new Date() && d <= new Date(Date.now()+7*86400000);
      }).length,
      custo_mes:           man.reduce((s:number,m:any)=>s+(m.custo_real??0),0),
      plano_urgente:       pl.length,
      score: Math.max(0, 100
        - eq.filter((e:any)=>e.status==="critico").length * 20
        - eq.filter((e:any)=>e.status==="atencao").length * 8
        - eq.filter((e:any)=>e.proxima_manutencao && new Date(e.proxima_manutencao)<new Date()).length * 12
      ),
    };
    res.json({ stats, equipamentos: eq, plano_urgente: pl });

});

  } catch (err) {

      analise: `Análise offline: verifique equipamentos críticos e manutenções vencidas. Acione o técnico responsável para os itens urgentes.`,
      resumo: { total: 0, criticos: 0, atencao: 0, vencidas: 0, proximas: 0, custo_mes: 0 },

});

// Esta função é chamada pelo cron de services/crons.ts

export async function verificarManutencaoAutomatica(condoId: string) {
  try {
    const { data: vencidos } = await getDB().from("equipamentos")
      .select("id,nome,categoria,local,proxima_manutencao,status")
      .eq("condominio_id", condoId)
      .lt("proxima_manutencao", new Date().toISOString().split("T")[0])
      .neq("status","inativo");

    if (!vencidos?.length) return;

    for (const eq of vencidos) {
      // Cria OS automática para manutenções vencidas há mais de 3 dias
      const diasAtraso = Math.floor(
        (Date.now() - new Date((eq as any).proxima_manutencao).getTime()) / 86400000
      );
      if (diasAtraso >= 3) {
        await criarOS(condoId, {
          titulo:     `Manutenção preventiva vencida: ${(eq as any).nome}`,
          prioridade: diasAtraso >= 7 ? "urgente" : "alta",
          area:       (eq as any).local ?? "Geral",
          sla_horas:  diasAtraso >= 7 ? 24 : 72,
        });

        // Atualiza status do equipamento
        await getDB().from("equipamentos").update({
          status: diasAtraso >= 7 ? "atencao" : "atencao",
          updated_at: new Date().toISOString(),
        }).eq("id", (eq as any).id);
      }
    }
  } catch {}
}
```
