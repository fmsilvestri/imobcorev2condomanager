```typescript
// Analisa contexto atual (OSs abertas, alertas MISP, próximas manutenções)
// Sugere tópico de comunicado relevante
// Botão "Di gera texto" → chama Claude API com contexto real
// Mostra: "Di sugere: baseado nas OSs abertas (elevador + piscina), enviar aviso de manutenção hoje"

async function gerarTextoDi(contexto: DiContexto): Promise<string> {
  const res = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 600,
      system: `Você é Di, síndica virtual do ${condominioNome} em Florianópolis.
               Escreva um comunicado profissional e amigável para os moradores.
               Máx 200 palavras. Tom: cordial, claro, sem jargão técnico.
               NÃO use markdown, asteriscos ou formatação especial.
               Termine sempre com: Atenciosamente, ${sindicoNome} - Síndico / Di - Síndica Virtual ImobCore`,
      messages: [{
        role: 'user',
        content: `Categoria: ${categoria}
Contexto: ${JSON.stringify(contexto)}
OSs abertas relevantes: ${JSON.stringify(ossAbertas)}
Escreva o comunicado completo.`
      }]
    })
  })
  const data = await res.json()
  return data.content[0].text
}
```

## 3. SERVIÇO DE ENVIO — `src/services/comunicadoService.ts`

```typescript
// ── WhatsApp via Z-API ──────────────────────────────────────────
async function enviarWhatsApp(config: CanalConfig, mensagem: string, destinatarios: string[]): Promise<WASendResult> {
  // Z-API endpoint: https://api.z-api.io/instances/{instanceId}/token/{token}/send-text
  // Formatar mensagem com emojis e quebras de linha
  // Enviar para lista de números cadastrados dos moradores
  // Retry automático para falhas
  
  for(const numero of destinatarios) {
    const res = await fetch(`https://api.z-api.io/instances/${config.wa_instance}/token/${config.wa_token}/send-text`, {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({phone: numero, message: mensagem})
    })
    // Registrar resultado por número
  }
}

// ── Telegram via Bot API ────────────────────────────────────────
async function enviarTelegram(config: CanalConfig, mensagem: string): Promise<TGSendResult> {
  // Telegram Bot API: https://api.telegram.org/bot{token}/sendMessage
  // Formatar com Markdown (negrito *texto*, itálico _texto_)
  // Enviar para chat_id do grupo do condomínio
  
  const res = await fetch(`https://api.telegram.org/bot${config.tg_bot_token}/sendMessage`, {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
      chat_id: config.tg_chat_id,
      text: formatarMarkdownTelegram(mensagem),
      parse_mode: 'Markdown'
    })
  })
  return res.json()
}

// ── Formatar para WhatsApp ──────────────────────────────────────
function formatarWhatsApp(titulo: string, corpo: string, condoNome: string): string {
  return `📢 *${condoNome}*\n\n*${titulo}*\n\n${corpo}\n\n_via ImobCore · Síndica Virtual Di_`
}

// ── Formatar para Telegram ──────────────────────────────────────
function formatarMarkdownTelegram(titulo: string, corpo: string): string {
  return `🏢 *Comunicado — Copacabana Beach Residence*\n\n*${titulo}*\n\n${corpo}\n\n_ImobCore v2 · @imobcore\_cb\_bot_`
}

// ── Envio multicanal ────────────────────────────────────────────
async function enviarMulticanal(comunicado: Comunicado, canais: string[], config: CanalConfig) {
  const resultados: Record<string, boolean> = {}
  
  if(canais.includes('whatsapp')) {
    const moradores = await supabase.from('moradores').select('telefone').eq('condominio_id', comunicado.condominio_id)
    const numeros = moradores.data?.map(m => m.telefone).filter(Boolean) || []
    const wa = await enviarWhatsApp(config, formatarWhatsApp(comunicado.titulo, comunicado.corpo, condoNome), numeros)
    resultados.whatsapp = wa.success
  }
  
  if(canais.includes('telegram')) {
    const tg = await enviarTelegram(config, formatarMarkdownTelegram(comunicado.titulo, comunicado.corpo))
    resultados.telegram = tg.ok
  }
  
  // Registrar resultado
  await supabase.from('comunicados').update({
    status: 'enviado',
    enviado_em: new Date().toISOString(),
    total_entregues: resultados.whatsapp ? numeros.length : 0
  }).eq('id', comunicado.id)
  
  return resultados
}
```

## 4. CLOUDFLARE WORKER — Automação por gatilho

Criar `workers/comunicado-trigger.ts`:

```typescript
// Recebe webhook do Supabase (realtime ou Postgres trigger)
// Avalia regras ativas em comunicado_regras
// Se gatilho === 'os_urgente_aberta' → gera texto com Di → envia WA + TG

export default {
  async fetch(req: Request, env: Env) {
    const body = await req.json()
    
    if(body.type === 'INSERT' && body.table === 'ordens_servico') {
      const os = body.record
      if(os.prioridade === 'urgente') {
        const config = await getCanaisConfig(os.condominio_id, env)
        const texto = await gerarTextoDiWorker(os, env.ANTHROPIC_KEY)
        await enviarTelegraBotWorker(config, texto)
        await enviarZAPIWorker(config, texto)
      }
    }
    
    return new Response('OK')
  }
}

async function gerarTextoDiWorker(os: any, apiKey: string) {
  const res = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {'Content-Type': 'application/json', 'x-api-key': apiKey, 'anthropic-version': '2023-06-01'},
    body: JSON.stringify({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 300,
      system: 'Você é Di, síndica virtual. Escreva um aviso conciso (máx 100 palavras) para moradores sobre manutenção urgente. Tom tranquilizador. Sem markdown.',
      messages: [{role: 'user', content: `OS urgente aberta: ${os.titulo}. Local: ${os.categoria}. Prestador: ${os.responsavel || 'a designar'}. Escreva o aviso.`}]
    })
  })
  const data = await res.json()
  return data.content[0].text
}
```
