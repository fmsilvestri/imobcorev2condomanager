# ImobCore · Totem Responsivo — Adaptação Automática de Tela
# Cole as 3 camadas no public/concierge/index.html

---

## POR QUE 3 CAMADAS?

```
CAMADA 1 — meta viewport   → diz ao browser "não escale — vou cuidar disso"
CAMADA 2 — CSS fluid       → font-size, espaçamentos e layout fluem com vw/vh
CAMADA 3 — JS scale engine → escala tudo proporcionalmente em telas grandes
```

Resultado:
  • iPhone 6 (375px)      → totem compacto e legível
  • Samsung Tab A8 (600px) → 1.5× maior, sem overflow
  • iPad Air (820px)       → 2× maior, usa bem o espaço
  • Monitor 1080p (1080px) → 2.7× maior, cinematográfico
  • TV 4K (3840px)         → 9× maior, sem pixelação (vetorial/rem)

## CAMADA 1 — <head>: substituir o meta viewport

```html
<head>
  <meta charset="UTF-8">

  <!--
    NÃO use maximum-scale=1 aqui — o JS de escala vai cuidar disso.
    initial-scale=1 garante que o browser não aplique zoom automático.
    width=device-width permite que o JS leia window.innerWidth corretamente.
  -->
  <meta name="viewport"
        content="width=device-width, initial-scale=1.0, viewport-fit=cover">

  <!-- Wake lock: impede o tablet de dormir (Chrome 84+) -->
  <meta name="mobile-web-app-capable"    content="yes">
  <meta name="apple-mobile-web-app-capable" content="yes">
  <meta name="apple-mobile-web-app-status-bar-style" content="black-translucent">
  <meta name="theme-color" content="#04040c">

  <title>Di · Concierge</title>
  <!-- [[TOTEM_CONFIG]] -->
  <link href="https://fonts.googleapis.com/css2?family=Nunito:wght@400;600;700;800;900&display=swap"
        rel="stylesheet">
```

## CAMADA 3 — JS: Scale Engine (colar no fim do `<script>`, antes do fechamento)

```javascript
// ══════════════════════════════════════════════════════════════════════════════
// TOTEM SCALE ENGINE — adapta o layout a qualquer tamanho de tela/tablet
// Estratégia: transform: scale() no #stage, centralizado no #stage-wrap
// Base de design: 390×720px. Em qualquer outra tela, escala proporcionalmente.
// ══════════════════════════════════════════════════════════════════════════════

(function TotemScaleEngine() {

  // Dimensões de design (base fixa — não alterar)
  const BASE_W = 390;
  const BASE_H = 720;

  // Elementos do DOM
  const stage = document.getElementById('stage');
  const wrap  = document.getElementById('stage-wrap');

  if (!stage || !wrap) {
    console.warn('[TotemScale] #stage ou #stage-wrap não encontrado.');
    return;
  }

  /**
   * Calcula e aplica a escala ótima para a tela atual.
   * Usa o menor fator entre largura e altura para garantir que
   * o conteúdo sempre cabe na tela sem overflow.
   */
  function aplicarEscala() {
    const vw = window.innerWidth;
    const vh = window.innerHeight;

    // Fator de escala: garante que o stage cabe sem cortar
    const scaleW = vw / BASE_W;
    const scaleH = vh / BASE_H;
    const scale  = Math.min(scaleW, scaleH);

    // Dimensões reais do stage após escala
    const realW = BASE_W * scale;
    const realH = BASE_H * scale;

    // Offset para centralizar o stage na tela
    const offsetX = (vw - realW) / 2;
    const offsetY = (vh - realH) / 2;

    // Aplicar transform: translate + scale
    stage.style.transform       = `translate(${offsetX}px, ${offsetY}px) scale(${scale})`;
    stage.style.transformOrigin = 'top left';
    stage.style.width           = BASE_W + 'px';
    stage.style.height          = BASE_H + 'px';

    // Fundo que preenche a tela toda (bordas pretas, como TV)
    wrap.style.background = '#000';

    // Ajustar font-size raiz proporcional à escala (reforça o rem system)
    // 16px × scale = font-size base escalada
    // clamp garante mínimo legível e máximo sensato
    const fs = Math.max(12, Math.min(16 * scale, 80));
    document.documentElement.style.fontSize = fs + 'px';

    // Log de diagnóstico (remover em produção)
    if (window.__TOTEM__?.debug) {
      console.log(`[TotemScale] ${vw}×${vh} → scale=${scale.toFixed(3)} | stage=${Math.round(realW)}×${Math.round(realH)} | fs=${fs.toFixed(1)}px`);
    }
  }

  // Aplicar na inicialização
  aplicarEscala();

  // Reaplicar no resize (tablet girado, janela redimensionada)
  let resizeTimer: ReturnType<typeof setTimeout>;
  window.addEventListener('resize', () => {
    clearTimeout(resizeTimer);
    resizeTimer = setTimeout(aplicarEscala, 50); // debounce 50ms
  });

  // Reaplicar na mudança de orientação (portrait ↔ landscape)
  window.addEventListener('orientationchange', () => {
    // Aguarda o browser terminar a rotação antes de calcular
    setTimeout(aplicarEscala, 200);
  });

  // Reaplicar quando o teclado virtual abrir/fechar (Android)
  if ('visualViewport' in window) {
    window.visualViewport!.addEventListener('resize', () => {
      clearTimeout(resizeTimer);
      resizeTimer = setTimeout(aplicarEscala, 50);
    });
  }

  // Expor função globalmente (útil para debug e recalibração manual)
  (window as any).totemRecalibrar = aplicarEscala;

})();
```

## CAMADA EXTRA — Wake Lock (tablet não dorme no modo totem)

```javascript
// Impede o tablet de bloquear a tela enquanto o totem está ativo.
// Wake Lock API — Chrome 84+, Edge 84+, Opera 71+.
// Em Safari e Firefox mais antigos, usa fallback de vídeo silencioso.

(async function TotemWakeLock() {
  let wakeLock: WakeLockSentinel | null = null;

  async function requestWakeLock() {
    // Tenta API nativa primeiro
    if ('wakeLock' in navigator) {
      try {
        wakeLock = await (navigator as any).wakeLock.request('screen');
        console.log('[WakeLock] Ativo — tela não vai dormir.');
        wakeLock.addEventListener('release', () => {
          console.log('[WakeLock] Liberado.');
        });
      } catch (err) {
        console.warn('[WakeLock] Falhou:', err);
        fallbackWakeLock();
      }
    } else {
      fallbackWakeLock();
    }
  }

  // Fallback: vídeo transparente em loop (truque clássico para iOS/Safari)
  function fallbackWakeLock() {
    const v = document.createElement('video');
    v.setAttribute('loop', '');
    v.setAttribute('muted', '');
    v.setAttribute('playsinline', '');
    v.setAttribute('autoplay', '');
    v.style.cssText = 'position:fixed;top:-1px;left:-1px;width:1px;height:1px;opacity:0.001;pointer-events:none';
    // Base64 de um vídeo WebM vazio (1×1 px, 1 frame, ~80 bytes)
    v.src = 'data:video/webm;base64,GkXfo59ChoEBQveBAULygQRC84EIQoKEd2VibUKHgQRChYECGFOAZwEAAAAAAAASE4mNm4EAAAAAAAAWVcYAAAAAAAAAAGcBAAAAAAAAPmCD';
    document.body.appendChild(v);
    v.play().catch(() => {});
    console.log('[WakeLock] Fallback via vídeo ativo.');
  }

  // Reativar quando a aba/app volta ao foco (wake lock some ao sair)
  document.addEventListener('visibilitychange', () => {
    if (document.visibilityState === 'visible') requestWakeLock();
  });

  await requestWakeLock();
})();
```
